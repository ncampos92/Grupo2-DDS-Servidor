1class ApplicationController < ActionController::Base
  
2  # Prevent CSRF attacks by raising an exception.
  
3  # For APIs, you may want to use :null_session instead.
  
4  #protect_from_forgery with: :null_session
  
5  include SessionHelper
  
6  helper_method :is_admin?, :is_editor?
  
7
  
8  #Verifica que el usuario sea vlido.
  
 9  def restrict_access
  
10    respond_to do |format|
  
11      format.html {
  
12        if !logged_in?
  
13          flash[:notice] = "Unauthorized access - please log in"
  
14          redirect_to root_url
  
15        end
  
16      }
  
17      format.json {
  
18        if !logged_in_api?
  
19          render json: {"error":{"description":"Not logged in - unauthorized access", "login_url":"/login"}}, status: :forbidden
  
20        end
  
21      }
  
22    end
  
23  end
  
24
  
25  #Indica si el usuario tiene nivel de acceso 'admin'
  
26  def is_admin?
  
27    if current_user || current_user_api
  
28      respond_to do |format|
  
29        format.html{
  
30          return current_user.nivel_acceso == 'admin'
  
31        }
  
32        format.json{
  
33          return current_user_api.nivel_acceso == 'admin'
  
34        }
  
35      end
  
36    end
  
37  end
  
38
  
39  #Indica si el usuario tiene nivel de acceso 'editor'
  
40  def is_editor?
  
41    if current_user || current_user_api
  
42      respond_to do |format|
  
43        format.html{
  
44          return current_user.nivel_acceso == 'editor'
  
45        }
  
46        format.json{
  
47          return current_user_api.nivel_acceso == 'editor'
  
48        }
  
49      end
  
50    end
  
51  end
